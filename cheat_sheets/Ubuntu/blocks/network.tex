\begin{block}{Network}%
	\begin{tabular}{p{\bashcolumnsize}p{\desccolumnsize}}
		\bcom{hostname}{Display system hostname}\\
		\bcom{ifconfig}{Show network information}\\
		\bcom{iwconfig}{Show wireless network information}\\
		\bcom{sudo iwlist scan}{Scan for wireless networks}\\
		\bcom{ping [host]}{Send ping to \texttt{[host]} and display the results}\\
		\bcom{whois [domain]}{Display whois information for \texttt{[domain]}}\\
		\bcom{dig [domain]}{Display DNS information for \texttt{[domain]}}\\
		\bcom{dig -x [host]}{Reverse lookup \texttt{[host]}}\\
		\bcom{wget [file]}{Download \texttt{[file]}}\\
		\bcom{wget -c [file]}{Continue stopped download of \texttt{[file]}}\\
		\bcom{scp [file] [host]}{Copy \texttt{[file]} to the directory \texttt{[host]} on a different machine}\\
		\bcom{rsync -a [/location] [/backup/]}{Sync the contents of a location with the backup directory}\\
	\end{tabular}
