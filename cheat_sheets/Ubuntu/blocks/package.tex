\begin{block}{Package management}%
	\begin{tabular}{p{\bashcolumnsize}p{\desccolumnsize}}
		\bcom{apt update}{Refresh package list}\\
		\bcom{apt upgrade}{Upgrade all packages}\\
		\bcom{apt dist-upgrade}{Upgrade Ubuntu version}\\
		\bcom{apt autoremove}{Remove all obsolete packages}\\
		\bcom{apt install [package]}{Install \texttt{[package]}}\\
		\bcom{apt remove [package]}{Remove \texttt{[package]}}\\
		\bcom{apt search [package]}{Search for \texttt{[package]} in package list}\\
		\bcom{dpkg -i [package].deb}{Manually install \texttt{[package]} using a .deb file}\\
	\end{tabular}
\end{block}
