\begin{block}{File management}%
	\begin{tabular}{p{\bashcolumnsize}p{\desccolumnsize}}
		\bcom{mkdir [dir]}{Create the directory \texttt{[dir]}}\\
		\bcom{touch [file]}{Create \texttt{[file]}}\\
		\bcom{cp [source] [target]}{Copy \texttt{[source]} to \texttt{[target]}}\\
		\bcom{cp -r [source] [target]}{Copy \texttt{[source]} to \texttt{[target]} and create directory for \texttt{[target]}}\\
		\bcom{cat [file]}{Outputs the contents of \texttt{[file]}}\\
		\bcom{cat >[file]}{Writes the standard input into \texttt{[file]}}\\
		\bcom{head [file]}{Outputs the first 10 lines of \texttt{[file]}}\\
		\bcom{tail [file]}{Outputs the last 10 lines of \texttt{[file]}}\\
		\bcom{tail -f [file]}{Continues to output lines as \texttt{[file]} grows}\\
		\bcom{rm [file]}{Deletes \texttt{[file]}}\\
		\bcom{rm -r [dir]}{Deletes \texttt{[dir]} and contained elements}\\
		\bcom{rsync -a [directory] [backup]}{Sync the contents of \texttt{[directory]} with the directory \texttt{[backup]}}\\
	\end{tabular}
\end{block}
