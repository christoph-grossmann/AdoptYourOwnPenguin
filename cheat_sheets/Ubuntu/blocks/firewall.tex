\begin{block}{Firewall}%
	\begin{tabular}{p{\bashcolumnsize}p{\desccolumnsize}}
		\bcom{ufw enable}{Turn on the firewall}\\
		\bcom{ufw disable}{Turn off the firewall}\\
		\bcom{ufw default allow}{Allow all connections by default}\\
		\bcom{ufw default deny}{Deny all connections by default}\\
		\bcom{ufw status}{Display current status and rules}\\
		\bcom{ufw allow [port]}{Allow traffic on \texttt{[port]}}\\
		\bcom{ufw deny [port]}{Block \texttt{[port]}}\\
		\bcom{ufw deny from [ip]}{Block traffic from \texttt{[ip]}}\\
	\end{tabular}
\end{block}
